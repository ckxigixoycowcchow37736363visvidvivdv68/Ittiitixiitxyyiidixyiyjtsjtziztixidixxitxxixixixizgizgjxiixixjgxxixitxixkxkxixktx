Tutorial of Color changing background

Step 1: Set up your Sketchware project

1.Open Sketchware and create a new project.

2.Add a LinearLayout (linear1) to your main screen. This layout will be animated by the color change.

3.In the Events section, find onCreate and insert the following code for add source directly block. 

  color1 = 0xffffffff;
anim = ObjectAnimator.ofInt(linear1, "backgroundColor", color1);
anim.setEvaluator(new ArgbEvaluator());
anim.setDuration(1000);
new Thread() {
public void run() {
while(true) {
try {
Thread.sleep(1000);
} catch (InterruptedException e) {
e.printStackTrace();
}
runOnUiThread(new Runnable() {
public void run() {
red2 = (int)(java.lang.Math.random() * 128 + 127);
green2 = (int)(java.lang.Math.random() * 128 + 127);
blue2 = (int)(java.lang.Math.random() * 128 + 127);
color2 = 0xff << 24 | (red2 << 16) | (green2 << 8) | blue2;
anim.setIntValues(color1, color2);
anim.start();
color1 = color2;
}
});
}
}
}.start();
}
